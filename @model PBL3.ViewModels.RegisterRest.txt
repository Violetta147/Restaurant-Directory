@section Scripts {
    @{
        await Html.RenderPartialAsync("_ValidationScriptsPartial");
    }
    <script>
        $(document).ready(function () {
            var showLoginModalFlag = @Html.Raw(Json.Serialize(ViewData["ShowLoginModal"] ?? false));
            var loginReturnUrl = @Html.Raw(Json.Serialize(ViewData["LoginReturnUrl"] as string ?? Url.Action("Create", "Restaurants"))); // Lấy từ ViewData, fallback nếu null

            if (showLoginModalFlag && !@User.Identity.IsAuthenticated.ToString().ToLower()) {
                console.log("Requesting login modal to be shown. ReturnUrl: " + loginReturnUrl);
                var urlToLoadLoginModal = "/Account/LoginModal?returnUrl=" + encodeURIComponent(loginReturnUrl);

                if ($('#loginModalInstance').length === 0) {
                    $('#loginModalPlaceholder').html('<div class="modal fade" id="loginModalInstance" tabindex="-1" aria-labelledby="loginModalLabel" aria-hidden="true"></div>');
                }
                var $modalInstance = $('#loginModalInstance');

                $modalInstance.load(urlToLoadLoginModal, function () {
                    var form = $("#loginFormModal");
                    if (form.length && typeof $.validator !== 'undefined' && typeof $.validator.unobtrusive !== 'undefined') {
                        form.removeData('validator');
                        form.removeData('unobtrusiveValidation');
                        $.validator.unobtrusive.parse(form);
                        console.log("Validation re-parsed for #loginFormModal in Restaurants/Create.");
                    }
                    var modal = new bootstrap.Modal($modalInstance[0]);
                    modal.show();
                });
            }
        });
    </script>
}
